  <script>
    const names = ["20, 60, 89-95", "50, 85, 86", "98 US", "29, 31", "98 CA", "35, 65", "30, 32, 33", "15", "13, 14", "9, 10", "12, 17", "DSD", "24, 25, 70, 85", "54, 75, 16, 80", "11, 55, 56"];

    const descriptions = {
      "20, 60, 89-95": "20, 60, 89-95.htm", // Replace with actual page URLs
      "24, 25, 70, 85": "24, 25, 70, 85.htm",
      "13, 14": "13, 14.htm",
      "29, 31": "29, 31.htm",
      "50, 85, 86": "50, 85, 86.htm",
      "98 US": "98 US.htm",
      "98 CA": "98 CA.htm",
      "35, 65": "35, 65.htm",
      "15": "15.htm",
      "54, 75, 16, 80": "54, 75, 16, 80.htm",
      "30, 32, 33": "30, 32, 33.htm",
      "9, 10": "9, 10.htm",
      "12, 17": "12, 17.htm",
      "11, 55, 56": "11, 55, 56.htm",
      "DSD": "DSD.htm"
    };

    function createButtons() {
      const nameList = document.getElementById("nameList");
      nameList.innerHTML = '';

      names.forEach(name => {
        const li = document.createElement("li");
        li.className = "item";

        const button = document.createElement("button");
        button.textContent = name;
        button.setAttribute("aria-expanded", "false");
        button.type = "button"; // Set button type

        button.addEventListener("click", () => {
          // Navigate to the corresponding page
          window.location.href = descriptions[name];
        });

        li.appendChild(button);
        nameList.appendChild(li);
      });
    }

    createButtons();
  </script>
